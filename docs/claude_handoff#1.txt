sim:
	verilator --binary --sv --trace \
		-o $(TOP) \
		$(RTL_DIR)/*.sv $(TB_DIR)/$(TOP).sv $(TB_ASSERT_DIR)/*.sv \
		--top-module $(TOP)
	./obj_dir/$(TOP) +verilator+seed+$(shell date +%s)
	gtkwave ../sim/$(TOP).vcd

synth:
	yosys -p "read_verilog -sv $(RTL_DIR)/$(MODULE).sv; \
	           synth -top $(MODULE); \
	           write_verilog ../sim/$(MODULE)_synth.v; \
	           stat"

cpu_synth:
	yosys -p "read_verilog -sv $(RTL_DIR)/*.sv; \
	           synth -top $(TOP); \
	           stat"

clean:
	rm -rf obj_dir
	rm -f ../sim/*.vcd

The CPU has just been pipelined with an ID/EX pipeline register. Current state:
systemverilog// riscv_cpu.sv
module riscv_cpu (
    input logic clk,
    input logic rst,
    output logic [31:0] pc,
    output logic [31:0] alu_result,
    output logic zero
);
    logic [4:0] rs1_addr, rs2_addr, rd_addr;
    logic [31:0] imm;
    logic [3:0] alu_op;
    logic reg_wr_en;
    logic [31:0] instr;
    logic alu_src;
    logic pc_src;
    logic [2:0] func3;
    logic branch;
    logic mem_read, mem_write, mem_to_reg;
    logic [31:0] dmem [255:0];
    logic [31:0] mem_data;
    logic [31:0] wb_data;
    logic [31:0] rs2_data;

    // ID/EX pipeline register signals
    logic [4:0]  id_ex_rs1_addr, id_ex_rs2_addr, id_ex_rd_addr;
    logic [31:0] id_ex_imm;
    logic [3:0]  id_ex_alu_op;
    logic        id_ex_reg_wr_en;
    logic        id_ex_alu_src;
    logic        id_ex_mem_read;
    logic        id_ex_mem_write;
    logic        id_ex_mem_to_reg;

    assign pc_src = branch && ((func3 == 3'b000 && zero) ||
                                (func3 == 3'b001 && !zero));

    always_ff @(posedge clk) begin
        if (rst) pc <= 32'd0;
        else     pc <= pc_src ? (pc + imm) : (pc + 32'd4);
    end

    // ID/EX pipeline register
    always_ff @(posedge clk) begin
        if (rst) begin
            id_ex_rs1_addr  <= 5'b0;
            id_ex_rs2_addr  <= 5'b0;
            id_ex_rd_addr   <= 5'b0;
            id_ex_imm       <= 32'b0;
            id_ex_alu_op    <= 4'b0;
            id_ex_reg_wr_en <= 1'b0;
            id_ex_alu_src   <= 1'b0;
            id_ex_mem_read  <= 1'b0;
            id_ex_mem_write <= 1'b0;
            id_ex_mem_to_reg<= 1'b0;
        end else begin
            id_ex_rs1_addr  <= rs1_addr;
            id_ex_rs2_addr  <= rs2_addr;
            id_ex_rd_addr   <= rd_addr;
            id_ex_imm       <= imm;
            id_ex_alu_op    <= alu_op;
            id_ex_reg_wr_en <= reg_wr_en;
            id_ex_alu_src   <= alu_src;
            id_ex_mem_read  <= mem_read;
            id_ex_mem_write <= mem_write;
            id_ex_mem_to_reg<= mem_to_reg;
        end
    end

    assign mem_data = dmem[alu_result[9:2]];
    assign wb_data  = id_ex_mem_to_reg ? mem_data : alu_result;

    always_ff @(posedge clk) begin
        if (id_ex_mem_write)
            dmem[alu_result[9:2]] <= rs2_data;
    end

    riscv_fetch_decode riscv_fetch_decode_inst(
        .clk(clk), .rst(rst), .pc(pc),
        .rs1_addr(rs1_addr), .rs2_addr(rs2_addr), .rd_addr(rd_addr),
        .imm(imm), .alu_op(alu_op), .reg_wr_en(reg_wr_en),
        .instr(instr), .alu_src(alu_src), .branch(branch), .func3(func3),
        .mem_read(mem_read), .mem_write(mem_write), .mem_to_reg(mem_to_reg)
    );

    riscv_execute riscv_execute_inst(
        .clk(clk), .rst(rst),
        .rs1_addr(id_ex_rs1_addr), .rs2_addr(id_ex_rs2_addr),
        .rd_addr(id_ex_rd_addr), .reg_wr_en(id_ex_reg_wr_en),
        .alu_src(id_ex_alu_src), .alu_op(id_ex_alu_op),
        .imm(id_ex_imm), .wb_data(wb_data),
        .alu_result(alu_result), .zero(zero), .rs2_data(rs2_data)
    );
endmodule

RV32I Quick Reference
Student has the full HTML reference table. Key encodings:
R-type: funct7[6:0] | rs2[4:0] | rs1[4:0] | funct3[2:0] | rd[4:0] | opcode[6:0]
I-type: imm[11:0]               | rs1[4:0] | funct3[2:0] | rd[4:0] | opcode[6:0]
S-type: imm[11:5]  | rs2[4:0]  | rs1[4:0] | funct3[2:0] | imm[4:0]| opcode[6:0]
B-type: imm[12|10:5]|rs2[4:0]  | rs1[4:0] | funct3[2:0] |imm[4:1|11]|opcode[6:0]
